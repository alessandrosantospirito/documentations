SS21

a) Filter by MAX

  • id und name aller Studierenden die das höchste Semester haben

xxxxxxxxxx
SELECT s.id, s.name
FROM students s
WHERE semester = (SELECT MAX(semester) FROM students);

b) Left Exclusive

  • id's von Aufgaben die noch nicht verwendet wurden (in grades)

xxxxxxxxxx
SELECT q.id
FROM questions q
LEFT OUTER JOIN grades g ON q.id = g.question_id
WHERE g.question_id IS NULL;

c)

  • IDs und Namen der Studierenden, die die Klausur mehrmals geschrieben haben

  • zusätzliche angabe des Jahres der letzten Klausur, an der sie teilgenommen haben

xxxxxxxxxx
SELECT s.id, s.name, MAX(g.year) AS last_exam_year
FROM students s
JOIN grades g ON g.student_id = s.id
GROUP BY s.id, s.name
HAVING COUNT(DISTINCT g.year) > 1
ORDER BY s.id;

d)

xxxxxxxxxx
SELECT name
FROM (
    SELECT name, COUNT(*) AS name_count
    FROM students
    GROUP BY name
) AS name_counts
WHERE name_count < 3;

e)

  • id und name der 10 studierenden die bei der Prüfung 2020 zu den besten gehörten

  • es darf davon ausgegangen werden dass mehr als 10 Studierende an dieser Prüfung teilnahmen

xxxxxxxxxx
SELECT s.id, s.name
FROM students s
JOIN grades g ON g.student_id = s.id
WHERE g.year = 2020
ORDER BY g.points DESC
LIMIT 10;

4

ii) SELECT * FROM G NATURAL JOIN Q

4 * 4

  • Natural Join entspricht dem Kartesischen Produkt, da es keine Übereinstimmung der Spalten gibt
