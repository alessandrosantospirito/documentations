GROUP BY

With GROUP BY we can guarantee that the grouped-elements are only displayed once and we can apply operations like MAX, COUNT on these grouped elements.

every column we want to select must be grouped with GROUP BY

xxxxxxxxxx
SELECT s_g.id, s_g.name, s_g.year
FROM students_grades as s_g
GROUP BY s_g.id, s_g.name, s_g.year;

MAX

xxxxxxxxxx
SELECT s_g.id, MAX(s_g.year)
FROM students_grades as s_g
GROUP BY s_g.id, s_g.year;

COUNT

xxxxxxxxxx
SELECT s_g.id,  COUNT(s_g.id)
FROM students_grades as s_g
GROUP BY s_g.id;

SUM

xxxxxxxxxx
SELECT s_i.buyer_id, SUM(s_i.buyer_id)
FROM stock_interest s_i
GROUP BY s_i.buyer_id

HAVING COUNT

xxxxxxxxxx
SELECT i.stock_id, COUNT(i.stock_id) AS quantity
FROM interest i
GROUP BY i.stock_id
HAVING COUNT(i.stock_id) > 1;

LIMIT

We can use LIMIT on a sorted table to get the x-th first entries.

xxxxxxxxxx
SELECT sorted_students.id, sorted_students.name
FROM (
    SELECT s_g.id, s_g.name, s_g.points
    FROM students_grades_2020 s_g
    ORDER BY s_g.points DESC
) AS sorted_students
LIMIT 10;
